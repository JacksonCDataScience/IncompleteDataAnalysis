n_sim <- 100

results <- data.frame(
  intercept_cc = numeric(n_sim),
  slope_cc = numeric(n_sim),
  var_cc = numeric(n_sim),
  cor_cc = numeric(n_sim),
  
  intercept_imp = numeric(n_sim),
  slope_imp = numeric(n_sim),
  var_imp = numeric(n_sim),
  cor_imp = numeric(n_sim)
)
for (i in 1:n_sim) {
  p_miss <- plogis(0.5 * scale(y))
  R <- rbinom(length(x), 1, p_miss)
  
  x_obs <- x
  x_obs[R == 1] <- NA
  data_obs <- data.frame(x = x_obs, y = y)
  
  fit_cc <- lm(y ~ x, data = data_obs)
  
  results$intercept_cc[i] <- coef(fit_cc)[1]
  results$slope_cc[i] <- coef(fit_cc)[2]
  results$var_cc[i] <- summary(fit_cc)$sigma^2
  results$cor_cc[i] <- cor(data_obs$x, data_obs$y, use = "complete.obs")
  
  fit_imp_xy <- lm(x ~ y, data = data_obs)
  
  x_pred <- predict(fit_imp_xy, newdata = data_obs)
  sigma <- summary(fit_imp_xy)$sigma
  
  x_imp <- x_obs
  missings <- is.na(x_imp)
  x_imp[missings] <- x_pred[missings] + rnorm(sum(missings), 0, sigma)
  
  data_imp <- data.frame(x = x_imp, y = y)
  
  fit_imp_yx <- lm(y ~ x, data = data_imp)
  
  results$intercept_imp[i] <- coef(fit_imp_yx)[1]
  results$slope_imp[i] <- coef(fit_imp_yx)[2]
  results$var_imp[i] <- summary(fit_imp_yx)$sigma^2
  results$cor_imp[i] <- cor(data_imp$x, data_imp$y)
}
